       # Get mouse press position

       self.mouse_down = True

   def mouseReleaseEvent(self, event: QMouseEvent):
       self.mouse_down = False

   def mouseMoveEvent(self, event: QMouseEvent) -> None:
       if self.mouse_down:
           # Calculate angle of rotation
           angle = self.calculate_angle(event.pos(), self.press_pos)

           # Rotate PIL image

           # Update UI

   def calculate_angle(self, pos, press_pos):
       # Calculate angle based on mouse delta
       return -(pos.x() - press_pos.x()) / (abs(pos.y() - press_pos.y()) + 1e-10)

   def apply_rotation(self, img, angle):
       # Rotate PIL image with the given angle
       rotated = img.rotate(angle)
       return rotated.resize(img.size, Image.BILINEAR)

        if self.image_path:
            image = Image.open(self.image_path)
            image.save('rotated_image.jpg')

def add_rotation_handle(self):
    if self.rotation_handle is not None:
        return  # –£–∂–µ –¥–æ–±–∞–≤–ª–µ–Ω–æ

    # –ö–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –¥–ª—è –∫—Ä—É–∂–∫–∞ (–Ω–µ–º–Ω–æ–≥–æ –≤—ã—à–µ —Ü–µ–Ω—Ç—Ä–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è)
    x = self.pixmap_item.boundingRect().center().x()
    y = self.pixmap_item.boundingRect().top() - 20  # –ù–µ–º–Ω–æ–≥–æ –≤—ã—à–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    radius = 10  # –†–∞–∑–º–µ—Ä –∫—Ä—É–∂–∫–∞

    # –°–æ–∑–¥–∞—ë–º –∫—Ä—É–∂–æ–∫ (—Ä—É—á–∫—É –¥–ª—è –≤—Ä–∞—â–µ–Ω–∏—è)
    self.rotation_handle = QGraphicsEllipseItem(-radius, -radius, 2 * radius, 2 * radius)
    self.rotation_handle.setBrush(Qt.blue)
    self.rotation_handle.setZValue(2)  # –ü–æ–≤–µ—Ä—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    self.rotation_handle.setPos(x, y)
    self.scene.addItem(self.rotation_handle)

    # –ì–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª—å–Ω–∞—è –ª–∏–Ω–∏—è –¥–ª—è —Å—Ä–∞–≤–Ω–µ–Ω–∏—è
    line_x1 = self.pixmap_item.boundingRect().left()
    line_x2 = self.pixmap_item.boundingRect().right()
    line_y = self.pixmap_item.boundingRect().top() + self.pixmap_item.boundingRect().height() / 2
    self.rotation_line = QGraphicsLineItem(line_x1, line_y, line_x2, line_y)
    self.rotation_line.setPen(Qt.red)
    self.scene.addItem(self.rotation_line)

    # –°–æ—Ö—Ä–∞–Ω—è–µ–º –¥–µ–π—Å—Ç–≤–∏–µ
    self.current_action = Action(type='rotate', value=0, final=False)
    self.add_action()

def mouseMoveEvent(self, event):
    if self.current_action is None:
        return
    if self.current_action.final:
        self.mouse_press_pos = None
        return

    # –í—Ä–∞—â–µ–Ω–∏–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –ø—Ä–∏ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–∏ –∫—Ä—É–∂–∫–∞

        center = self.pixmap_item.mapToScene(self.pixmap_item.boundingRect().center())  # –¶–µ–Ω—Ç—Ä –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        new_pos = event.scenePos()  # –ü–æ–ª—É—á–∞–µ–º –Ω–æ–≤—É—é –ø–æ–∑–∏—Ü–∏—é –º—ã—à–∏ –≤ —Å—Ü–µ–Ω–µ

        # –í—ã—á–∏—Å–ª—è–µ–º —É–≥–æ–ª –º–µ–∂–¥—É —Ü–µ–Ω—Ç—Ä–æ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏ –ø–æ–∑–∏—Ü–∏–µ–π –º—ã—à–∏

        dy = new_pos.y() - center.y()
        angle = math.degrees(math.atan2(dy, dx))  # –£–≥–æ–ª –≤ –≥—Ä–∞–¥—É—Å–∞—Ö

        # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —É–≥–æ–ª –ø–æ–≤–æ—Ä–æ—Ç–∞ –¥–ª—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
        self.pixmap_item.setRotation(angle)

        # –ü–µ—Ä–µ–º–µ—â–∞–µ–º rotation_handle –≤ –Ω–æ–≤—É—é –ø–æ–∑–∏—Ü–∏—é
        radius = 20  # –†–∞–¥–∏—É—Å –æ—Ç —Ü–µ–Ω—Ç—Ä–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –¥–æ —Ä—É—á–∫–∏

        handle_y = center.y() + radius * math.sin(math.radians(angle))
        self.rotation_handle.setPos(handle_x, handle_y)

from PIL import Image

def crop_image_with_negative_x(input_file, output_file, crop_params):
    """
    –û–±—Ä–µ–∑–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω–æ–π –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã x.

    :param input_file: –ò–º—è –≤—Ö–æ–¥–Ω–æ–≥–æ —Ñ–∞–π–ª–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    :param output_file: –ò–º—è –≤—ã—Ö–æ–¥–Ω–æ–≥–æ —Ñ–∞–π–ª–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    :param crop_params: –ö–æ—Ä—Ç–µ–∂ (x, y, width, height)
    """
    # –û—Ç–∫—Ä—ã–≤–∞–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
    image = Image.open(input_file)
    width, height = image.size
    x, y, crop_width, crop_height = crop_params

    # –ï—Å–ª–∏ x –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–π, –¥–æ–±–∞–≤–ª—è–µ–º —Å–ª–µ–≤–∞ –ø—É—Å—Ç—É—é –æ–±–ª–∞—Å—Ç—å
    if x < 0:
        # –°–æ–∑–¥–∞–µ–º –Ω–æ–≤–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ —Å —É–≤–µ–ª–∏—á–µ–Ω–Ω–æ–π —à–∏—Ä–∏–Ω–æ–π
        new_width = width + abs(x)
        new_image = Image.new("RGB", (new_width, height), (255, 255, 255))  # –ë–µ–ª—ã–π —Ñ–æ–Ω
        new_image.paste(image, (abs(x), 0))  # –í—Å—Ç–∞–≤–ª—è–µ–º –∏—Å—Ö–æ–¥–Ω–æ–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ —Å–ø—Ä–∞–≤–∞
        image = new_image
        x = 0  # –¢–µ–ø–µ—Ä—å x –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è —Å 0

    # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ–±—ã –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –Ω–µ –≤—ã—Ö–æ–¥–∏–ª–∏ –∑–∞ –≥—Ä–∞–Ω–∏—Ü—ã –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    x = max(0, x)
    y = max(0, y)
    crop_width = min(crop_width, image.width - x)
    crop_height = min(crop_height, image.height - y)

    # –í—ã–ø–æ–ª–Ω—è–µ–º –æ–±—Ä–µ–∑–∫—É
    cropped_image = image.crop((x, y, x + crop_width, y + crop_height))

    # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
    cropped_image.save(output_file)
